PROJECT(IKAlgorithms)

SET(CMAKE_MODULE_PATH ${IKAlgorithms_SRCS})

ADD_EXECUTABLE(IKAlgorithms ${IKAlgorithms_SRCS})
TARGET_LINK_LIBRARIES(IKAlgorithms BTKIO ik)
